\section{Motivation}
% ============================================================================

\begin{frame}{Heavy Neutral Lepton — what we're searching for}
  \footnotesize
  Right-handed sterile $\nu$ mixing with SM $\nu$ via parameter $V$ (Type-I seesaw).\\
  VBF production: 2 quarks exchange $W$, radiating 2 forward jets.\\
  Decay: $N\to \ell W^{(\ast)}\to \ell jj$ or $\ell\ell\nu$.

  \vspace{2mm}
  Final state under study: $\mu^{\pm} e^{\mp}$ + 2 VBF jets.\\
  Signal kinematics:
  \begin{itemize}\setlength{\itemsep}{0pt}\footnotesize
    \item $\mu$ from $W$: $p_T>26$\,GeV (trigger), $>45$\,GeV (signal)
    \item $e$ from $N$: $p_T \gtrsim 5$\,GeV — \emph{soft}
    \item 2 jets: $|\Delta\eta_{jj}|>3$, $m_{jj}>400$\,GeV
    \item OS lepton pair, no $b$-jets, $0<m_{\ell\ell}<150$\,GeV
  \end{itemize}

  \concl{Soft electrons + jet-rich topology $\Rightarrow$ background dominated by
  prompt-lepton processes \emph{and} a non-trivial fake-electron component.
  Latter cannot be modelled by MC and requires a data-driven estimate.}
\end{frame}

\begin{frame}{Backgrounds in the $\mu e$ channel}
  \footnotesize
  \begin{tabular}{lll}
    \toprule
    Process & Type & Why it lands in our SR \\
    \midrule
    $t\bar t$ (di-lepton)   & prompt + flip & W-decay leptons; OS-natural, SS via flip \\
    Single top              & prompt + fake & W-decay $\mu$, $b\to e$ semi-lep \\
    $Z\to\tau\tau$          & prompt        & true $\mu e$ from $\tau$ decays \\
    $WW$, $WZ$, $ZZ$        & prompt        & multilepton $\to \mu e$ + lost lepton \\
    VBF $H\to WW^\ast$      & prompt        & $H\to\mu\nu e\nu$ — VBF jets natural \\
    $W$+jets                & fake          & $W\to\mu\nu$ + jet-fake $e$ \\
    QCD multijet            & fake          & both leptons from semi-leptonic decays \\
    \bottomrule
  \end{tabular}

  \motiv{Prompt processes are well-modelled in MC and subtracted MC vs. data.\\
  Multijet + W+jets jet-fake-$e$ are NOT well-modelled $\Rightarrow$ data-driven.}
\end{frame}

\begin{frame}{What is a ``fake'' electron?}
  \footnotesize
  Reconstructed $e$ that is \emph{not} from a prompt $W,Z,H$ decay.\\
  In MC tagged via \fname{el\_truth\_isPrompt} and \fname{el\_IFFClass}.

  \vspace{2mm}
  \textbf{Fake-electron sources}:\\[0.5mm]
  \begin{tabular}{lll}
    \toprule
    Source & Mechanism & Typical $|d_0/\sigma|$ \\
    \midrule
    Heavy flavor ($b\to e\nu X$, IFF=8/9) & semi-leptonic $b/c$ decay & 1.5--4 (displaced)\\
    Photon conversion (IFF=5) & $\gamma\to e^+e^-$ in tracker & varies, often large \\
    Light-flavor mis-ID (IFF=10) & $\pi^0\to\gamma\gamma$, hadronic shower & $<1$ (PV-like) \\
    $\tau$ decay (IFF=7) & $\tau\to e\nu\bar\nu$ & medium \\
    \bottomrule
  \end{tabular}

  \concl{Sources have very different ID/iso behaviour. We \emph{measure} their
  combined acceptance from data $\Rightarrow$ Asymptotic Matrix Method (AMM).}
\end{frame}

\begin{frame}{Why a same-sign control region for fakes}
  \footnotesize
  At leading order, prompt SM processes producing $\mu e$ are
  charge-conjugation symmetric: $\mu^+ e^-$ and $\mu^- e^+$ in equal
  parts, but no $\mu^\pm e^\pm$ without one fake.

  \vspace{2mm}
  \textbf{Same-sign $\mu e$ event} is therefore overwhelmingly:
  \begin{itemize}\setlength{\itemsep}{0pt}
    \item A fake (non-prompt) electron — \emph{the dominant case}, OR
    \item A prompt $e$ with mis-measured charge — small, MC-subtracted.
  \end{itemize}

  After prompt-MC subtraction, the SS data residual is a clean
  fake-enriched sample.

  \motiv{$b$-veto in the CR removes $t\bar t$ (which has real $b$-jets) and
  enriches the W+jets-like fake population that dominates the SR fakes.}
\end{frame}

\section{The Asymptotic Matrix Method}
% ============================================================================

\begin{frame}{The matrix equation}
  \footnotesize
  Loose population = real + fake, tight = $\varepsilon_r N_R + \varepsilon_f N_F$:
  \[
    \begin{pmatrix}N_T\\N_L\end{pmatrix}
    =
    \begin{pmatrix}\varepsilon_r & \varepsilon_f\\
                   1-\varepsilon_r & 1-\varepsilon_f\end{pmatrix}
    \begin{pmatrix}N_R\\N_F\end{pmatrix}
  \]
  Inverting and recovering tight-fake yield:
  \[
    N_F^{\text{tight}} = \varepsilon_f \cdot
       \frac{N_T - \varepsilon_r N_L}{\varepsilon_f - \varepsilon_r}
  \]
\end{frame}

\begin{frame}{Per-event AMM weights}
  \footnotesize
  Instead of solving for totals, weight each event:
  \[
    w_T = \frac{\varepsilon_f(\varepsilon_r-1)}{\varepsilon_r-\varepsilon_f}
        \;<\;0\quad\text{(tight events)}
  \]
  \[
    w_L = \frac{\varepsilon_f\,\varepsilon_r}{\varepsilon_r-\varepsilon_f}
        \;>\;0\quad\text{(loose-only events)}
  \]
  $\sum_{\text{events in region}} w_{T/L} \;=\; N_F^{\text{tight}}$ in that region.

  \motiv{Per-event weight branches let downstream tools (TRExFitter, plotting)
  apply systematics without re-running the matrix inversion.\\
  Output: \fname{AMM/data\_with\_electron\_fake\_weights.root} → branches
  \fname{fake\_weight\_*}.}
\end{frame}

\section{Fake Efficiency $\varepsilon_f$}

\begin{frame}{Measuring $\varepsilon_f$ in the SS $\mu e$ CR}
  \footnotesize
  Data-driven, prompt-MC subtracted:
  \[
    \varepsilon_f(p_T,|\eta|) =
       \frac{N_T^{\text{data}} - N_T^{\text{MC prompt}}}
            {N_L^{\text{data}} - N_L^{\text{MC prompt}}}

  Charge-misID SF applied: \fname{el\_charge\_misid\_effSF\_tight\_NOSYS}.

  \motiv{MC subtraction removes prompt-$e$ contamination of the SS CR
  (charge-flipped Z$\to$ee, residual prompt-from-WW, etc.) so that
  data $-$ MC is the pure fake population.}

  \vspace{1mm}
  Macros: \fname{efficiency/MC\_eff.C}, \fname{data\_eff.C},
  \fname{calculate\_fake\_rates.C}.

\section{Systematic Uncertainties}
% ============================================================================

\begin{frame}{Stat — concept}
  \motiv{Per-bin Poisson error from limited SS CR data statistics.
  Largest at high $p_T$ where the CR is sparse.}
  \footnotesize
  Macro: \fname{build\_fake\_systematics.C} → \fname{fake\_rate\_statUp/Down}.\\
  Propagated as bin-by-bin uncorrelated NP (one per (pT,$|\eta|$) bin) at fit time.
\end{frame}

\begin{frame}{MET dependence — concept}
  \motiv{$\varepsilon_f$ is measured inclusively in MET, but applied to events
  with a specific MET distribution (SR, VR, ttbar/Zττ CRs). Variation in
  $\varepsilon_f$ between low-/high-MET sub-regions becomes an NP.}
  \footnotesize
  Method: split SS CR at $E_T^\text{miss}=50$\,GeV; up/down maps come from
  the larger of the two splits. Macro: \fname{build\_fake\_systematics.C}
  $\Rightarrow$ \fname{fake\_rate\_MET\_up/down}.
\end{frame}

\begin{frame}{Composition — concept}
  \motiv{The fake source mix in the inclusive CR may differ from the
  VBF-tagged SR (different jet kinematics, different $b$/$c$ fraction).
  Reweight CR-fake-source mix to the SR-fake-source mix and re-derive
  $\varepsilon_f$.}
  \footnotesize
  Symmetric formula (avoids unbounded blow-ups):
  \[
    f_c(p_T,|\eta|) =
       2\,\frac{\varepsilon_f^{\text{SR}}-\varepsilon_f^{\text{CR}}}
                {\varepsilon_f^{\text{SR}}+\varepsilon_f^{\text{CR}}}
  \]
  Macros: \fname{build\_composition\_systematic.C}
  $\to$ \fname{build\_fake\_systematics.C}
  $\Rightarrow$ \fname{fake\_rate\_comp\_up/down}.
\end{frame}

\begin{frame}{Prompt-subtraction — concept}
  \motiv{$\varepsilon_f$ depends on the prompt-MC subtraction.
  If MC normalisation is wrong by $\pm 20\%$, $\varepsilon_f$ shifts.
  Vary $k$ (the prompt-MC scale factor) in
  $\varepsilon_f = (N_T - k\cdot N_T^{\text{MC}})/(N_L - k\cdot N_L^{\text{MC}})$
  by $\pm 0.2$.}

\section*{B.2 — Investigation: where did the high-$p_T$ $\varepsilon_f$ come from?}
% ============================================================================

\begin{frame}{Trigger-bias study — the structural finding}
  \motiv{First $\varepsilon_f$ rose to $\sim 0.5$ at high $p_T$ — unphysical.
  Single-electron HLT requires near-tight $e$ at L1 → if applied to the CR,
  the loose population is HLT-biased toward tight-passing.}
  \footnotesize
  Before: \fname{passTrigger = mu \char`\|\char`\| e}, $\varepsilon_f^{[50,200]}\!\approx\!0.4$.\\
  After: \fname{passTrigger = mu only}, $\varepsilon_f^{[50,200]}\!\approx\!0.20$--$0.30$.
  \concl{Electron-trigger removal restored a physically reasonable shape.
  Standard ATLAS recipe (HNL multilep arXiv:2204.11138). Now uniform in
  measurement macros; SR/VR keep full mu$\lor$e for max acceptance.}
\end{frame}

\begin{frame}{$p_T$ binning iterations}
  \motiv{Initial bins $\{15,25,35,50,80,200\}$ followed standard ATLAS
  practice but had a stat-limited high-$p_T$ tail; we tested several
  variations to find a stable map.}
  \scriptsize
  \begin{tabular}{ll}
    \toprule
    Binning & Outcome \\
    \midrule
    $\{15,25,35,50,80,200\}$ & last bin stat-limited; flat-100\% syst flagged \\
    $\{15,25,35,50,200\}$    & merged tail; better stats; central for a while \\
    $\{5,15,25,35,50,60,200\}$ & added low-$p_T$+transition; \fname{[50,60]} broke (LF=444\%) \\
    \textbf{$\{5,15,25,35,50,200\}$} & current: keeps low-$p_T$, drops broken transition \\
    \bottomrule
  \end{tabular}
  \concl{Final binning: \fname{kPtBins = \{5,15,25,35,50,200\}}.}
\end{frame}

\begin{frame}{Is rising $\varepsilon_f$ at high $p_T$ a bug or physics?}
  \motiv{$\varepsilon_f(p_T)$: 0.08 $\to$ 0.28. Tested four hypotheses to
  decide whether to ``fix'' it or accept it.}
  \footnotesize
  \begin{itemize}\setlength{\itemsep}{0pt}
    \item Charge-flip contamination $\to$ ruled out (SF $\sim$ 1.0, backup)
    \item Multijet (anti-iso $\mu$ + MET-split) $\to$ ruled out
    \item Heavy-flavor real-$e$ leakage $\to$ ruled out (truth + d0sig + $\Delta R$)
    \item AMM vs. fake-factor fragility $\to$ ruled out (1.02--1.15 ratio)
  \end{itemize}
  \concl{All four ruled out. LF jet$\to e$ dominates at every $p_T$;
  high-$p_T$ LF passes tight more often $\Rightarrow$ \emph{rising
  $\varepsilon_f$ is real physics, not a bug.}}
\end{frame}
