\section{Motivation}
% ============================================================================

\begin{frame}{Heavy Neutral Lepton — what we're searching for}
  \footnotesize
  Right-handed sterile $\nu$ mixing with SM $\nu$ via $V$ (Type-I seesaw).
  VBF production: 2 quarks exchange $W$, radiating 2 forward jets.
  Decay: $N\!\to\!\ell W^{(\ast)}\!\to\!\ell jj$ or $\ell\ell\nu$.

  \vspace{1mm}
  Final state: $\mu^{\pm} e^{\mp}$ + 2 VBF jets.
  \begin{itemize}\setlength{\itemsep}{0pt}\footnotesize
    \item $\mu$ from $W$: $p_T>26$ trigger / $>45$\,GeV signal
    \item $e$ from $N$: $p_T \gtrsim 5$\,GeV — \emph{soft}
    \item 2 jets: $|\Delta\eta_{jj}|>3$, $m_{jj}>400$\,GeV
    \item OS, no $b$-jets, $0<m_{\ell\ell}<150$\,GeV
  \end{itemize}

  \concl{Soft electrons + jet-rich topology $\Rightarrow$ background dominated
  by prompt-lepton processes \emph{and} a non-trivial fake-electron component
  that requires a data-driven estimate.}
\end{frame}

\begin{frame}{Backgrounds in the $\mu e$ channel}
  \footnotesize
  \begin{tabular}{lll}
    \toprule
    Process & Type & Why it lands in our SR \\
    \midrule
    $t\bar t$ (di-lep)   & prompt + flip & $W$ leptons; OS-natural, SS via flip \\
    Single top           & prompt + fake & $W\!\to\!\mu$, $b\!\to\!e$ \\
    $Z\!\to\!\tau\tau$   & prompt        & true $\mu e$ from $\tau$ decays \\
    $WW/WZ/ZZ$           & prompt        & multilepton $\to \mu e$+lost lep \\
    VBF $H\!\to\!WW^\ast$& prompt        & $H\!\to\!\mu\nu e\nu$ + VBF jets \\
    $W$+jets             & fake          & $W\!\to\!\mu\nu$ + jet-fake $e$ \\
    QCD multijet         & fake          & both leptons from semi-lep decays \\
    \bottomrule
  \end{tabular}
  \motiv{Prompt processes are well-modelled in MC and subtracted MC vs.\ data.
  Multijet + W+jets jet-fake-$e$ are NOT well-modelled $\Rightarrow$ data-driven.}
\end{frame}

\begin{frame}{What is a ``fake'' electron?}
  \footnotesize
  Reconstructed $e$ that is \emph{not} from a prompt $W,Z,H$ decay.
  Tagged via \fname{el\_truth\_isPrompt}, \fname{el\_IFFClass}.

  \vspace{1mm}
  \begin{tabular}{lll}
    \toprule
    Source & Mechanism & Typical $|d_0/\sigma|$ \\
    \midrule
    Heavy flavor (IFF=8/9) & $b/c\!\to\!e\nu X$ & 1.5--4 (displaced) \\
    Photon conversion (IFF=5) & $\gamma\!\to\!e^+e^-$ & varies \\
    Light-flavor mis-ID (IFF=10) & $\pi^0$, hadronic & $<1$ (PV-like) \\
    $\tau$ decay (IFF=7) & $\tau\!\to\!e\nu\bar\nu$ & medium \\
    \bottomrule
  \end{tabular}
  \concl{Sources have different ID/iso behaviour. We \emph{measure} their
  combined acceptance from data $\Rightarrow$ Asymptotic Matrix Method (AMM).}
\end{frame}

\begin{frame}{Why a same-sign control region}
  \footnotesize
  At LO, prompt $\mu e$ is charge-conjugation symmetric: $\mu^+ e^-$ and $\mu^- e^+$
  in equal parts, but no $\mu^\pm e^\pm$ without one fake.

  \vspace{1mm}
  \textbf{SS $\mu e$ event} is therefore overwhelmingly:
  \begin{itemize}\setlength{\itemsep}{0pt}
    \item A fake (non-prompt) electron — \emph{the dominant case}, OR
    \item A prompt $e$ with mis-measured charge — small, MC-subtracted.
  \end{itemize}

  After prompt-MC subtraction, SS data is a clean fake-enriched sample.

  \motiv{$b$-veto in the CR removes $t\bar t$ (which has real $b$-jets) and
  enriches the W+jets-like fake population that dominates the SR fakes.}
\end{frame}

\section{The Asymptotic Matrix Method}
% ============================================================================

\begin{frame}{The Asymptotic Matrix Method}
  \footnotesize
  Loose population = real + fake; tight = $\varepsilon_r N_R + \varepsilon_f N_F$:
  \[
    \begin{pmatrix}N_T\\N_L\end{pmatrix}
    =
    \begin{pmatrix}\varepsilon_r & \varepsilon_f\\
                   1{-}\varepsilon_r & 1{-}\varepsilon_f\end{pmatrix}
    \begin{pmatrix}N_R\\N_F\end{pmatrix}
    \;\Longrightarrow\;
    N_F^{\text{tight}}=\sum_{\text{events}} w_{T/L}
  \]
  \[
    w_T = \frac{\varepsilon_f(\varepsilon_r{-}1)}{\varepsilon_r{-}\varepsilon_f}\;<0
    \quad\quad
    w_L = \frac{\varepsilon_f\,\varepsilon_r}{\varepsilon_r{-}\varepsilon_f}\;>0
  \]

  Per-event weights stored as branches in
  \fname{AMM/data\_with\_electron\_fake\_weights.root} for downstream tools.

  \motiv{Per-event weight ntuple lets TRExFitter / plotting apply systematics
  by switching \fname{fake\_weight\_nominal} $\to$ \fname{fake\_weight\_<syst>\_up/down}
  with no re-running the matrix inversion.}
\end{frame}

\begin{frame}{Measuring $\varepsilon_f$ and $\varepsilon_r$}
  \footnotesize
  \[
    \varepsilon_f(p_T,|\eta|) =
       \frac{N_T^{\text{data}} - N_T^{\text{MC prompt}}}
            {N_L^{\text{data}} - N_L^{\text{MC prompt}}}
    \quad\quad
    \varepsilon_r(p_T,|\eta|) = \frac{N_T^{\text{MC prompt}}}{N_L^{\text{MC prompt}}}
  \]
  \begin{itemize}\setlength{\itemsep}{0pt}\footnotesize
    \item $\varepsilon_f$: in SS $\mu e$ CR, prompt-MC subtracted, charge-misID-SF weighted
    \item $\varepsilon_r$: in OS prompt CR, MC truth-matched ($l1$, $l2$ both prompt)
    \item Both: muon-only triggering, $|\eta|<2.47$, crack-veto, $b$-veto
  \end{itemize}
  Macros: \fname{efficiency/MC\_eff.C}, \fname{data\_eff.C}, \fname{real\_eff.C},
  \fname{calculate\_fake\_rates.C}.
\end{frame}

\begin{frame}{What systematic NPs and why}
  \footnotesize
  Four NPs propagated to the fake estimate:
  \begin{tabular}{lll}
    \toprule
    NP & Source & Why we keep it \\
    \midrule
    Stat & per-bin Poisson on data CR & dominant at high $p_T$ where stats fall off \\
    MET  & low-MET vs.\ high-MET split & 3.7$\sigma$ resolved bin pull (backup) \\

\section*{B.2 — Investigation: where did high-$p_T$ $\varepsilon_f$ come from?}
% ============================================================================

\begin{frame}{Trigger-bias study — the structural finding}
  \motiv{First $\varepsilon_f$ rose to $\sim 0.5$ at high $p_T$ — unphysical.
  Single-electron HLT requires near-tight $e$ at L1 → loose population
  HLT-biased toward tight-passing.}
  \footnotesize
  Before: \fname{passTrigger = mu$\lor$e}, $\varepsilon_f^{[50,200]}\!\approx\!0.4$.\\
  After:  \fname{passTrigger = mu only}, $\varepsilon_f^{[50,200]}\!\approx\!0.20$--$0.30$.
  \concl{Electron-trigger removal restored a physically reasonable shape.
  Standard ATLAS recipe — uniform across measurement macros.}
\end{frame}

\begin{frame}{$p_T$ binning iterations}
  \motiv{Initial $\{15,25,35,50,80,200\}$ followed standard ATLAS but had a
  stat-limited tail. Tested several variations to find a stable map.}
  \scriptsize
  \begin{tabular}{ll}
    \toprule
    Binning & Outcome \\
    \midrule
    $\{15,25,35,50,80,200\}$ & last bin stat-limited; flat-100\% syst flagged \\
    $\{15,25,35,50,200\}$    & merged tail; better stats \\
    $\{5,15,25,35,50,60,200\}$ & added low-$p_T$ + transition; \fname{[50,60]} broke \\
    \textbf{$\{5,15,25,35,50,200\}$} & current — keeps low-$p_T$, drops broken transition \\
    \bottomrule
  \end{tabular}
  \concl{Final binning: \fname{kPtBins = \{5,15,25,35,50,200\}}.}
\end{frame}

\begin{frame}{Is rising $\varepsilon_f$ a bug or physics?}
  \motiv{Tested four hypotheses to decide whether to ``fix'' or accept it.}
  \footnotesize
  \begin{itemize}\setlength{\itemsep}{0pt}
    \item Charge-flip contamination $\to$ ruled out (SF $\sim$ 1.0)
    \item Multijet (anti-iso $\mu$ + MET-split) $\to$ ruled out
    \item Heavy-flavor real-$e$ leakage $\to$ ruled out (truth + d0sig + $\Delta R$)
    \item AMM vs.\ fake-factor fragility $\to$ ruled out (1.02--1.15 ratio)
  \end{itemize}
  \concl{All four ruled out. LF jet$\to e$ dominates at every $p_T$;
  high-$p_T$ LF passes tight more often $\Rightarrow$
  \emph{rising $\varepsilon_f$ is real physics, not a bug.}}
\end{frame}
